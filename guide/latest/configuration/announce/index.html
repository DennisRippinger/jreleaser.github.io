Announce

Releases can be announced to several media once they have been successfully created.

Legend:

  • required

  • optional

  • may use environment variable

  • accepts Name Templates

  • YAML

  • JSON

  • Maven

  • Gradle

# 
announce:
  # Enable or disable all configured announcers.
  # Defaults to `true`.
  # 
  enabled: true
{
  // 
  "announce": {
    // Enable or disable all configured announcers.
    // Defaults to `true`.
    // 
    "enabled": true
  }
}
<jreleaser>
  <!--
    
  -->
  <announce>
    <!--
      Enable or disable all configured announcers.
      Defaults to `true`.
      
    -->
    <enabled>true</enabled>
  </announce>
</jreleaser>
jreleaser {
  // icon;dot-circle[]
  announce {
    // Enable or disable all configured announcers.
    // Defaults to `true`.
    // 
    enabled = true
  }
}
